**타이머 API**

- Callback

  ```javascript
  const delay = (wait, callback) => {
    setTimeout(callback, wait);
  }
  ```

  ```javascript
  function runCallback() {
    resetTitle();
    playVideo();
    delay(1000, () => {
      pauseVideo();
      displayTitle();
      delay(500, () => {
        highlightTitle();
        delay(2000, resetTitle);
      });
    });
  }
  ```

- Promise Constructor

  ```javascript
  const sleep = (wait) => {
    return new Promise((resolve) => {
      setTimeout(()=>{
        resolve('hello');
      }, wait);
    });
  }
  ```

  ```javascript
  function runPromise() {
    resetTitle();
    playVideo();
  
    sleep(1000).then((param) => {
      console.log(param); //hello
      pauseVideo();
      displayTitle();
      return "world";
    })
      .then((param) => {
        console.log(param); //world
        sleep(500);
      })
      .then(highlightTitle)
      .then(sleep.bind(null, 2000))
      .then(resetTitle)
      .catch(err => {
        console.log(err);
      })
  }
  ```

- async/await

  ```javascript
  async function runAsync() {
    resetTitle();
    playVideo();
  
    let test = await sleep(1000);
    console.log(test); //hello
  
    pauseVideo();
    displayTitle();
    await sleep(500);
    highlightTitle();
    await sleep(2000);
    resetTitle();
  }
  ```

**fs 모듈**

- Callback

  ```javascript
  const fs = require("fs");
  const getDataFromFile = function (filePath, callback) {
    fs.readFile(filePath, 'utf8', (err, data) => {
      if(err)
        callback(err, null);
      else
        callback(null, data);
      });
  };
  getDataFromFile('README.md', (err, data) => {
      console.log(data)
    }
  );
  
  module.exports = {
    getDataFromFile
  };
  ```

- Promise Constructor

  ```javascript
  const fs = require("fs");
  const getDataFromFilePromise = filePath => {
    return new Promise((resolve, rejects) => {
      fs.readFile(filePath, 'utf8', (err, data) => {
        if (err) rejects(err);
        resolve(data);
      }
      )
    })
  };
  getDataFromFilePromise('README.md').then(data => console.log(data));
  
  module.exports = {
    getDataFromFilePromise
  };
  ```

- Basic Chaining (.then으로 이어서 Chaining)

  ```javascript
  const path = require('path');
  const { getDataFromFilePromise } = require('./02_promiseConstructor'); //위의 함수 사용하기 위함
  
  const user1Path = path.join(__dirname, 'files/user1.json');//객체 1개 들어있음
  const user2Path = path.join(__dirname, 'files/user2.json');//객체 1개 들어있음
  
  const readAllUsersChaining = () => {
    var result = [];
      return new Promise((resolve) => {
        
     getDataFromFilePromise(user1Path)
      .then(data => {
        result.push(JSON.parse(data)); //result배열에 string형인 data를 JSON.parse를 사용하여 객체로 변환 후 넣어줌
        return getDataFromFilePromise(user2Path)
      }).then(data => {
        result.push(JSON.parse(data)); //위와 마찬가지로 넣어줌
        resolve(result); //result를 resolve로 return 해줌
      })
    });
  }
  readAllUsersChaining();
  
  module.exports = {
    readAllUsersChaining
  }
  ```

- Promise.all 사용

  ```javascript
  const path = require('path');
  const { getDataFromFilePromise } = require('./02_promiseConstructor');
  
  const user1Path = path.join(__dirname, 'files/user1.json');
  const user2Path = path.join(__dirname, 'files/user2.json');
  
  const readAllUsers = () => {
    var result = [];
    return Promise.all([getDataFromFilePromise(user1Path), getDataFromFilePromise(user2Path)])
      .then(data => { //Promise.all을 사용하면 return 값을 data에 배열로 가져옴
        for (var i = 0; i < data.length; i++) {
          result.push(JSON.parse(data[i]));
        }
        return result
      })
  }
  readAllUsers()
  
  module.exports = {
    readAllUsers
  }
  ```

- async/await

  ```javascript
  const path = require('path');
  const { getDataFromFilePromise } = require('./02_promiseConstructor');
  
  const user1Path = path.join(__dirname, 'files/user1.json');
  const user2Path = path.join(__dirname, 'files/user2.json');
  
  const readAllUsersAsyncAwait = async () => {
    result = [ ];
  
    result.push(JSON.parse(await getDataFromFilePromise(user1Path)));
    result.push(JSON.parse(await getDataFromFilePromise(user2Path)));
    
    return result
  }
  readAllUsersAsyncAwait();
  
  module.exports = {
    readAllUsersAsyncAwait
  }
  ```

**fetch API**

- Basic Chaining (.then으로 이어서 Chaining)

  ```javascript
  function getNewsAndWeather() {
    const newsURL = 'http://localhost:4999/data/latestNews';
    const weatherURL = 'http://localhost:4999/data/weather';
    
    return new Promise((resolve, rejects) => {
      fetch(newsURL)
        .then((response) => response.json())
        .then((json_news) => {
          fetch(weatherURL)
            .then((response) => response.json())
            .then((json_weather) => {
             var obj = {
                news: json_news.data,
                weather: json_weather
              }
              resolve(obj);
            })
        })
        .catch((error) => console.log(error))
    })
  }
  getNewsAndWeather();
  
  if (typeof window === 'undefined') {
    module.exports = {
      getNewsAndWeather
    }
  }
  ```

- Promise.all 사용

  const newsURL = 'http://localhost:4999/data/latestNews';
  const weatherURL = 'http://localhost:4999/data/weather';
  
  function getNewsAndWeatherAll() {
